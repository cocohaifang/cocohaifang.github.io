2.查询被锁记录，并杀进程解锁社保工作日志汇总(20120330).xls
select sess.sid,
sess.serial#,
lo.oracle_username,
lo.os_user_name,
ao.object_name,
lo.locked_mode
from v$locked_object lo,
dba_objects ao,
v$session sess
where ao.object_id = lo.object_id and lo.session_id = sess.sid;

3.查询消耗CPU比较高的SQL处理办法
通过topas 找出占用CPU比例高的进程，通过以下SQL找出对应SQL
Select s.machine,s.program,l.* from v$sql l ,v$session s ,v$process p
where l.sql_id=s.sql_id
And s.paddr=p.addr
And p.spid=’33554556’

3.查看SQL执行进度
SELECT SE.SID,
OPNAME,
TRUNC(SOFAR / TOTALWORK 100, 2) || ‘%’ AS PCT_WORK,
ELAPSED_SECONDS ELAPSED,
ROUND(ELAPSED_SECONDS (TOTALWORK - SOFAR) / SOFAR) REMAIN_TIME,
SQL_TEXT
FROM V$SESSION_LONGOPS SL, V$SQLAREA SA, V$SESSION SE
WHERE SL.SQL_HASH_VALUE = SA.HASH_VALUE
AND SL.SID = SE.SID
AND SOFAR != TOTALWORK
ORDER BY START_TIME
